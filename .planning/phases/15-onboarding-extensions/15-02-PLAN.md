    // CONF-05: warn owner if existing session bookings exist.
    const sessions = await listSessions(business.id);

      return (

        'Οι πελάτες θα βλέπουν πλέον μόνο τις ορισμένες σεζόν.'
      );
    }
  }

  await setBookingMode(business.id, mode);
